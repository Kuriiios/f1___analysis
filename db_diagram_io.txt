Table event_round {
  event_round_id Integer [primary key]
  round_number smallint
  country string(30)
  location string(30)
  event_name string(30)
  event_date datetime
  sprint_event boolean
  soft smallint
  medium smallint
  hard smallint
}

Table session {
  session_id integer [primary key]
  event_round_id Integer 
  session_name string(20)
  session_date datetime
}

Table driver {
  driver_id integer [primary key]
  driver_number smallint
  driver_abbreviation string(3)
  driver_name string(40)
  driver_color string(7)
  country string(2)
}

Table team {
  team_id integer [primary key]
  team_name string(30)
  team_abbreviation string(3)
  team_color string(7)
}

Table driver_team_assignment {
  driver_team_assignment_id integer [primary key]
  session_id integer
  driver_id integer
  team_id integer
}

Table weather {
  weather_id Integer [primary key]
  session_id integer
  air_temp smallint
  humidity smallint
  pressure smallint
  rainfall boolean
  track_temp smallint
  wind_direction smallint
  time_ smallint
}

Table lap {
  lap_id Integer [primary key]
  driver_team_assignment_id Integer
  compound String(15)
  laptime_s integer
  lap_number smallint
  stint smallint
  pit_in_time float
  pit_out_time float
  sector1_time float
  sector2_time float
  sector3_time float
  sector1_session_time float
  sector2_session_time float
  sector3_session_time float
  speed_i1 smallint
  speed_i2 smallint
  speed_fl smallint
  speed_st smallint
  is_personal_best boolean
  tyre_life smallint
  lap_start_time float
  lap_start_date datetime
  position smallint
  track_status smallint
  deleted boolean
}

Table car_data {
  car_data_id integer
  lap_id integer
  date_time datetime
  time_ float
  session_time float
  rpm integer
  speed integer
  n_gear integer
  throttle integer
  brake boolean
  drs smallint
  track_status smallint
  distance float
  differential_distance float
  relative_distance float
  distance_driver_ahead integer
}

Table pos_data {
  pos_data_id integer
  lap_id integer
  date_time datetime
  time_ float
  session_time float
  x smallint
  y smallint
  z smallint
  car_status boolean
  track_status smallint
}

Ref : event_round.event_round_id < session.event_round_id

Ref : team.team_id < driver_team_assignment.team_id
Ref : session.session_id < driver_team_assignment.session_id
Ref : weather.session_id < session.session_id 
Ref : lap.driver_team_assignment_id > driver_team_assignment.driver_team_assignment_id
